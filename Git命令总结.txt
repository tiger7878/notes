1、克隆远程仓库中的代码
	git clone xxx.git
2、添加所有文件监控
	git add .
3、提交到本地仓库
	git commit -m '说明'
4、提交到远程仓库
	git push -u origin master
5、本地创建并检出dev分支
	git checkout -b dev
6、查看分支
	git branch
7、提交并创建远程分支dev
	git push --set-upstream origin dev
8、查看git配置信息
	git config --list
9、查看git用户名
	git config user.name
10、查看邮箱配置
	git config user.email
11、全局配置用户名
	git config --global user.name 'monkey'
12、全局配置邮箱
	git config --global user.email 'zhcw63@163.com'
13、从远程仓库克隆某个分支
	git clone -b branch_name git://xxx/project.git
14、切换到指定分支，并更新工作区
	git checkout branch-name
15、合并指定分支到当前分支
	git merge branch
16、git版本查看
	git --version
